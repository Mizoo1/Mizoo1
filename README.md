Ich habe auch ein unterhaltsames Snake-Spiel erstellt! Versuche die Schlange zu steuern und so viele Früchte wie möglich zu fressen.

Anweisungen:
1. Öffne den Screenshot [hier](https://example.com/path/to/screenshot.png), um das Snake-Spiel zu sehen.
2. Führe das Spiel lokal auf deinem Computer aus, indem du den Code aus der Datei "snake.html" verwendest.

```html
<!DOCTYPE html>
<html>
<head>
  <title>Snake-Spiel</title>
  <style>
    /* Stil für das Spielfeld */
    #game-board {
      width: 300px;
      height: 300px;
      border: 1px solid black;
      position: relative;
    }

    /* Stil für die Schlange und den Apfel */
    .snake-part, .apple {
      width: 10px;
      height: 10px;
      position: absolute;
    }

    .snake-part {
      background-color: green;
    }

    .apple {
      background-color: red;
    }
  </style>
</head>
<body>
  <h1>Snake-Spiel</h1>
  <div id="game-board"></div>
  <script>
    const gameBoard = document.getElementById('game-board');
    const snakeLength = 3;
    const snakeSpeed = 200; // Millisekunden zwischen den Bewegungen der Schlange
    let snake = [{ x: 150, y: 150 }];
    let apple = { x: 0, y: 0 };
    let direction = 'right';
    let changingDirection = false;

    function createApple() {
      apple.x = Math.floor(Math.random() * 30) * 10;
      apple.y = Math.floor(Math.random() * 30) * 10;
      const newApple = document.createElement('div');
      newApple.className = 'apple';
      newApple.style.left = `${apple.x}px`;
      newApple.style.top = `${apple.y}px`;
      gameBoard.appendChild(newApple);
    }

    function drawSnake() {
      snake.forEach((part, index) => {
        const snakePart = document.createElement('div');
        snakePart.className = 'snake-part';
        snakePart.style.left = `${part.x}px`;
        snakePart.style.top = `${part.y}px`;
        gameBoard.appendChild(snakePart);

        // Die erste Teile der Schlange (Kopf) wird etwas anders gestylt
        if (index === 0) {
          snakePart.style.backgroundColor = 'darkgreen';
          snakePart.style.borderRadius = '50%';
        }
      });
    }

    function clearBoard() {
      gameBoard.innerHTML = '';
    }

    function moveSnake() {
      const newHead = { x: snake[0].x, y: snake[0].y };

      switch (direction) {
        case 'up':
          newHead.y -= 10;
          break;
        case 'down':
          newHead.y += 10;
          break;
        case 'left':
          newHead.x -= 10;
          break;
        case 'right':
          newHead.x += 10;
          break;
      }

      snake.unshift(newHead);

      // Prüfe, ob die Schlange den Apfel gegessen hat
      if (newHead.x === apple.x && newHead.y === apple.y) {
        createApple();
      } else {
        snake.pop();
      }

      changingDirection = false;
    }

    function handleKeyDown(event) {
      if (changingDirection) return;
      changingDirection = true;

      const keyPressed = event.key;
      const oppositeDirections = { up: 'down', down: 'up', left: 'right', right: 'left' };

      if (keyPressed === oppositeDirections[direction]) return;

      direction = keyPressed;
    }

    createApple();
    drawSnake();

    setInterval(() => {
      clearBoard();
      moveSnake();
      drawSnake();
    }, snakeSpeed);

    document.addEventListener('keydown', handleKeyDown);
  </script>
</body>
</html>
